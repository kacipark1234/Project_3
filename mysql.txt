

CREATE DATABASE `Portfolio_3`;
USE `Portfolio_3`; 

CREATE TABLE `message`(     
    `id` int PRIMARY KEY AUTO_INCREMENT,  
    `author` VARCHAR(30),  
    `subject` text,  
    `content` text,  
    `date` datetime
);
CREATE TABLE `reply_message`(     
    `id` int PRIMARY KEY AUTO_INCREMENT,  
    `author` VARCHAR(30),  
    `subject` text,  
    `content` text,  
    `date` datetime,
    `reply_id` int
    
);
